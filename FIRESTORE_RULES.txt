rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isSignedIn() {
      return request.auth != null;
    }
    
    function getUserRole() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role;
    }
    
    function isAdmin() {
      return isSignedIn() && getUserRole() == 'admin';
    }
    
    function isTeacher() {
      return isSignedIn() && getUserRole() in ['teacher', 'admin'];
    }
    
    function isStudent() {
      return isSignedIn() && getUserRole() == 'student';
    }
    
    // User profiles - users can read own profile, admins manage all
    match /users/{userId} {
      allow read: if isSignedIn() && (request.auth.uid == userId || isAdmin());
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Students collection - stored by UID
    // Admins: full access
    // Teachers: can read all students
    // Students: can only read their own profile
    match /students/{uid} {
      allow read: if isAdmin() || isTeacher() || (isStudent() && request.auth.uid == uid);
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Active Sessions - for monitoring who is online
    // Admins: can read all sessions
    // Each user: can read/write only their own session
    match /activeSessions/{uid} {
      allow read: if isAdmin() || request.auth.uid == uid;
      allow write: if request.auth.uid == uid;
      allow create: if request.auth.uid == uid;
      allow delete: if request.auth.uid == uid || isAdmin();
    }
    
    // Logs - admins can read, all signed-in users can write
    match /logs/{logId} {
      allow read: if isAdmin();
      allow write: if isSignedIn();
      allow create: if isSignedIn();
    }
    
    // Attendance records - teachers and admins can read/write, students cannot
    match /attendance/{attendanceId} {
      allow read: if isAdmin() || isTeacher();
      allow write: if isAdmin() || isTeacher();
      allow create: if isAdmin() || isTeacher();
    }
    
    match /attendance/{attendanceId}/details/{detailId} {
      allow read: if isAdmin() || isTeacher();
      allow write: if isAdmin() || isTeacher();
    }
    
    // Mentors - teachers and admins can read, only admins can write
    match /mentors/{mentorId} {
      allow read: if isAdmin() || isTeacher();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Teams - all authenticated users can read, only admins can write
    match /teams/{teamId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Team Scores - all authenticated users can read (for rankings), only admins can write
    match /teamScores/{scoreId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    // Announcements - all authenticated users can read, admins can write
    match /announcements/{announcementId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Syllabus - all authenticated users can read, admins can write
    match /syllabus/{syllabusId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Rankings - all authenticated users can read
    match /rankings/{rankingId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    // App Config - admins only
    match /appConfig/{configId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
    }
    
    // System Health - admins only
    match /systemHealth/{healthId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
    }
    
    // Registration Requests - admins only
    match /registrationRequests/{requestId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }

    // Shared Schedules - admins create share links, anyone with link can read
    match /shared-schedules/{scheduleId} {
      allow read: if true; // Public read for share links
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isAdmin();
    }
    
    // Default deny for any other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
